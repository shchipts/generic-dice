[docs]def emissions_land_use_co2(n_points):
    """AFOLU emissions (GtCO2)."""
    e_points = np.zeros(n_points)
    e_points[0] = 2.6

    for x_points in range(1, n_points):
        e_points[x_points] = e_points[x_points - 1] * (1 - 0.115)

    return e_points
